### Button example usage

```tsx
import { Button } from '@/components/extendui/button';
```

```tsx
<Button variant={'bouncing'} tooltipText={`I'm bouncing!`}>
  Button
</Button>
```

    <tbody>
      <tr>
        <td>variant</td>
        <td>
          default | destructive | outline | secondary | ghost | link | ringHover
          | glowingRing | shimmer | bouncing
        </td>
        <td>
          Defines the visual style of the button. It supports several predefined
          styles like default, outline, etc.
        </td>
      </tr>
      <tr>
        <td>size</td>
        <td>default | sm | lg | icon</td>
        <td>
          Determines the size of the button. Available sizes are sm, lg, and
          icon.
        </td>
      </tr>
      <tr>
        <td>asChild</td>
        <td>boolean</td>
        <td>
          If true, the button will render as a child component using a Slot,
          useful for passing custom components.
        </td>
      </tr>
      <tr>
        <td>loading</td>
        <td>boolean</td>
        <td>If true, shows a loading spinner instead of the button content.</td>
      </tr>
      <tr>
        <td>loadingText</td>
        <td>string</td>
        <td>
          Text displayed next to the loading spinner when the button is in a
          loading state.
        </td>
      </tr>
      <tr>
        <td>tooltipText</td>
        <td>string</td>
        <td>If provided, shows a tooltip when hovering over the button.</td>
      </tr>
      <tr>
        <td>className</td>
        <td>string</td>
        <td>Additional CSS classes to apply custom styling to the button.</td>
      </tr>
      <tr>
        <td>children</td>
        <td>React.ReactNode</td>
        <td>The content inside the button, typically text or an icon.</td>
      </tr>
      <tr>
        <td>ref</td>
        <td>React.Ref&lt;HTMLButtonElement&gt;</td>
        <td>A ref to the button element for direct DOM manipulation.</td>
      </tr>
      <tr>
        <td>...props</td>
        <td>React.ButtonHTMLAttributes</td>
        <td>
          Additional props that can be passed to the button, such as onClick,
          disabled, etc.
        </td>
      </tr>
    </tbody>
  </table>
</div>
